		public int getLargest(){
		int total = 0;
    		for (int[] rowArray : matrix)
     			for (int item : rowArray)
				if (item > total)
        				total = item;
    		return total;
	}
		public int getColTotal(int key){
		int total = 0;
    		for (int[] rowArray : matrix)
     			for (int col = 0; col < rowArray.length; col++){
				if (col == key)
        				total += rowArray[col];
		}		
    		return total;
}
